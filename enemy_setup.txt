enemy setup

if any enemy routine pointer = 0 then it is ignored

balloon: {
    property = speed
        XYYY, where
        X = pixels per frame
        YYY = 0.YYY0
        
    palette bitmask high byte (properties 2)
        timer for respawn
        ff00 = 00ff frames
}

duct: {
    palette bitmask high byte (properties 2)
        $8000 bit = floor or ceiling duct
        
    property
        00ff
        low byte = room index to travel to
        
        ff00
        high byte = output x position
}

battery: {
    property
        amount of battery time
}

switch: {
    property
        data for target (its property)
    property2 (enemypal)
        $0200 bit = spritemap selection
    property3
        %rrrrrrrrssiiiiit
         
        r = room index of target
        s = property select of target
        t = type select: enemy (0) or object(1)
        i = index of target
        
        table for s:
        
        for objects:
            dw !objvariable,        0       00
               !objproperty,        2       40
               !objpal              4       80
               
        for enemies:
            dw !enemyproperty,      0       00
               !enemyproperty2,     2       40
               !enemyproperty3      4       80

    enemyvariable:
        spritemap select
            0000 = on
            0002 = off

}

all prizes: {
    (battery, extra life, clock, bands, foil)
    property3:
        item index bit
        each room gets one word, so a room can only have 16 prizes
        each item gets one bit and it must be unique
}

copter: {
    property:
        x speed.subspeed [speed is signed, subspeed is not]
            ssbb
            ss = speed
            bb = bb00 subspeed
    property2:
        high byte:
            %tttttttd
            d = select left or right
            t = respawn timer
        low byte:
            sprite properties
            
    property3:
        y speed.subspeed:
            ssbb
            ss = speed
            bb = bb00 subspeed
}
